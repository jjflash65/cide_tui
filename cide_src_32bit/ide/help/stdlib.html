---------------------------------------------------------
stdlib.h
---------------------------------------------------------

Die Bibliothek stdlib.h beinhaltet verschiedene
Funktionen zum Umgang mit Zahlen sowie Funktionen
zur Speicherverwaltung.

Hier aufgefuehrte Beispielprogramme benoetigen
zur Verdeutlichung die printf-Funktionalitaet.
Aus diesem Grund sind die aufgefuehrten Beispiel-
programme NICHT auf einem ATtiny2313 lauffaehig sondern
benoetigen min. einen ATmega8 !!!

Damit die Beispielprogramme zu stdlib.h auf einem
AVR-Controller nachvollzogen werden koennen, ist die
Datei rs232_io.c mittels den Linkeroptionen hinzu-
zufuegen. Soll ein Programm als Linux-PC Programm
ausgefuehrt werden, so ist die Headerdatei 'rs232_io.h'
sowie die Initialisierung der RS232 - rs232_init() -
und die Endlosschleife - while(1); - am Ende des
Beispielprogramms zu entfernen.

abs
---------------------------------------------------------


Das Makro abs berechnet den absoluten Betrag
einer Integer-Zahl.
  int abs(int x);
Prototyp in math.h (fuer real) / stdlib.h

Beispiel:
 #include <stdio.h>
 #include <math.h>
 #include <stdlib.h>
 #include "../include/rs232_io.h"
 
 int main(void)
 {
   int zahl = -1234;
 
   rs232_init();
 
   printf("Zahl: %d  Absolutwert: %d\n",    \
           zahl, abs(zahl));
   while(1);
 }

atof
---------------------------------------------------------


Konvertiert einen String in eine Gleitkommazahl.
  double atof(const char *s);
Prototyp in math.h / stdlib.h

atof liefert das Ergebnis der Interpretation bzw. den
Wert 0, wenn die Konvertierung nicht fehlerfrei ausge-
fuehrt werden konnte.

Um das nachfolgende Beispielprogramm ausfuehren zu koennen,
muss bei Verwendung mit AVR-Controllern in den Linker-
optionen die Auswahl 'floating point' aktiviert werden.

Beispiel:
 #include <stdlib.h>
 #include <stdio.h>
 #include "../include/rs232_io.h"
 
 int main(void)
 {
    float f;
    char *str = "12345.67";
 
    rs232_init();
 
    f = atof(str);
    printf("Text = %s Kommazahl = %f\n", str, f);
    while(1);
 }

bsearch
---------------------------------------------------------


Binaeres Absuchen einer Liste.
 void *bsearch(const void *key,
       const void *base, size_t *nelem,
       size_t width, int (*fcmp)(const void*,
                     const void*));
Prototyp in stdlib.h

Liefert die Adresse des ersten Elements von base,
dessen Inhalt mit dem von key uebereinstimmt. Wenn
die Suche erfolglos bleibt, ist das Ergebnis NULL.

bsearch interpretiert den Rueckgabewert der (benutzer-
definierten) Vergleichsfunktion *fcmp wie folgt:
  <  0 bedeutet: *elem1  < *elem2
  == 0 bedeutet: *elem1 == *elem2
  >  0 bedeutet: *elem1  > *elem2
Die Tabelle muss in aufsteigender Reihenfolge sortiert
sein.

Beispiel:
 #include <stdlib.h>
 #include <stdio.h>
 #include "../include/rs232_io.h"
 
 #define NELEMS(arr) (sizeof(arr) / sizeof(arr[0]))
 
 int numarray[] = {123, 145, 512, 627, 800, 933};
 
 int numeric (const int *p1, const int *p2)
 {
    return(*p1 - *p2);
 }
 
 int lookup(int key)
 {
    int *itemptr;
 
    /* Cast von (int(*)(const void *,const void*))
       wird benoetigt um einen 'type mismatch' Fehler
       beim kompilieren zu verhindern */
        
    itemptr = bsearch (&key, numarray, NELEMS(numarray),
              sizeof(int), 
              (int(*)(const void *,const void *))numeric);
              
    return (itemptr != NULL);
 }
 
 int main(void)
 {
 
    rs232_init();
 
    if (lookup(512))
       printf("512 ist in der Tabelle.\n\r");
    else
       printf("512 ist nicht in der Tabelle.\n\r");
 
    while(1);
 }

calloc
---------------------------------------------------------


Reserviert Platz im Hauptspeicher.
  void *calloc(size_t nitems, size_t size);
Prototyp in stdlib.h

Reserviert einen Block von nitems * size Bytes und
initialisiert sie mit 0.

Liefert einen Zeiger auf den neu belegten Block. Wenn nicht genug Platz vorhanden ist oder nitems oder size den Wert 0 hat, ist das Ergebnis NULL.

Beispiel:
 #include <stdio.h>
 #include <string.h>
 #include <stdlib.h>
 #include "../include/rs232_io.h"
 
 int main(void)
 {
    char *str = NULL;
 
    rs232_init();
 
    // Speicher fuer einen String reservieren (allozieren) 
    str = calloc(10, sizeof(char));
 
    strcpy(str, "Hallo");
 
    printf("Der Text ist: %s\n\r", str);
 
    free(str)                // Speicher freigeben    
 
    while(1);
 }

div
---------------------------------------------------------


dividiert zwei Integerwerte.
  div_t  div(int numer, int denom);
Prototyp in stdlib.h

div dividiert die als numer (Dividend) und als denom (Divisor)
uebergebenen Integerwerte und liefert eine Struktur des Typs div_t
zurueck. Dieser Datentyp ist in stdlib.h wie folgt definiert:
  typedef struct {
     int quot;  // Quotient (Divisionsergebnis)
     int rem;   // Divisionsrest 
  } div_t;
Beispiel:
 #include <stdlib.h>
 #include <stdio.h>
 #include "../include/rs232_io.h"
 
 int main(void)
 {
 
    div_t x;
 
    rs232_init();
 
    x = div(10,3);
    printf("10 div 3 = %d Rest %d\n", x.quot, x.rem);
 
    while(1);
 }

ltoa
---------------------------------------------------------


konvertiert einen long-Wert in einen String.
  char *ltoa(long value, char *string,
             int radix);
Prototyp in stdlib.h

radix definiert die Basis, bezueglich der value konvertiert
wird. radix muss groesser als 1 und kleiner als 37 sein.
radix=10 erzeugt dezimale Darstellung, radix=16
hexadezimale.

Beispiel:
 #include <stdlib.h>
 #include <stdio.h>
 #include "../include/rs232_io.h"
 
 int main(void)
 {
    char string[25];
    long zahl = 123456789L;
 
    rs232_init();
 
    ltoa(zahl,string,10);
    printf("Zahl: %ld  Text: %s\n", zahl, string);
 
    while(1);
 }

malloc
---------------------------------------------------------


dynamische Belegung von Speicherplatz.
  void *malloc(size_t size);
Prototyp in stdlib.h

size gibt die Groesse des zu reservierenden Speichers in Bytes
an. malloc liefert einen Zeiger auf den neu belegten Block bzw.
den Wert NULL, wenn nicht genuegend Platz auf dem Heap vorhanden
ist oder size den Wert 0 hat.

Beispiel:
 #include <stdio.h>
 #include <string.h>
 #include <stdlib.>
 #include "../include/rs232_io.h"
  
 int main(void)
 {
    char *str;
 
    rs232_init();
 
    // Speicher fuer einen String reservieren (allozieren)
    if ((str = malloc(10)) == NULL)
    {
       printf("Nicht genug Speicher vorhanden...\n\r");
       while(1);              // Programm anhalten
    }
 
    strcpy(str, "Hallo");
 
    printf("Der Text ist: %s\n",str);
 
    free(str);                // Speicher freigeben
 
    while(1);
 }

qsort
---------------------------------------------------------


sortiert die Elemente eines Arrays mit dem QuickSort
Algorithmus.
   void qsort(void *base, size_t nelem,
              size_t width, int(*fcmp)
              (const void *, const void *));
Prototyp in stdlib.h

nelem = Elementenzahl des Arrays base, width = Groesse eines
Elements. Die Routine fcmp muss ein Vergleichsergebnis
(<0, 0, >0) zurueckliefern.

Der Rueckgabewert von fcmp wird so interpretiert:
   < 0 bedeutet *elem1  < *elem2
   = 0 bedeutet *elem1 == *elem2
   > 0 bedeutet *elem1  > *elem2

Beispiel:
  #include <stdio.h>
  #include <stdlib.h>
  #include <string.h>
  #include "../include/rs232_io.h"
 
  int sort_function( const void *a, const void *b)
  {
     return( strcmp(a,b) );
  }
 
  int main(void)
  {
 
     char list[5][8] = { "Maier", "Mueller", "Schulz ",
		                 "Schmidt", "Albers" };
     int  x;
 
     rs232_init();
 
     qsort((void *)list, 5, sizeof(list[0]), sort_function);
     for (x = 0; x < 5; x++)
        printf("%s\n\r", list[x]);
     while(1);
  }

rand
---------------------------------------------------------


liefert eine "Zufallszahl" zurueck.
   int rand(void);
Prototyp in stdlib.h

Das Ergebnis liegt im Bereich von 0...RAND_MAX. RAND_MAX ist in
stdlib.h definiert als 0x7fff.

Beispiel:
 #include <stdlib.h>
 #include <stdio.h>
 #include "../include/rs232_io.h"
 
 int main(void)
 {
    int i;
 
    rs232_init();
 
    printf("10 Zufallszahlen von 0 bis 99\n\n\r");
    for(i=0; i<10; i++)
       printf("%d\n\r", rand() % 100);
    while(1);
 }

realloc
---------------------------------------------------------


aendert die Groesse eines dynamisch belegten Speicherblocks.
  void *realloc(void *block, size_t size);
Prototyp in stdlib.h

Liefert die neue (und evtl. veraenderte) Adresse des Blocks
(mit size Bytes) zurueck bzw. den Wert NULL, wenn der Block
aus irgendeinem Grund nicht modifiziert werden konnte (oder
size den Wert 0 hat).

Beispiel:
 #include <stdlib.h>
 #include <string.h>
 #include <stdio.h>
 #include "../include/rs232_io.h"
 
 int main(void)
 {
   char *str;
 
   rs232_init();
 
   str = malloc(10);      // Speicher reservieren (allozieren)
 
   strcpy(str, "Hallo");
 
   printf("Text : %s\n\r  Addresse : %p\n\r", str, str);
   str = realloc(str, 20);
   printf("Text : %s\n\r  Neue Adresse : %p\n\r", str, str);
 
   free(str);            // Speicher freigeben
 
   while(1);
 }
